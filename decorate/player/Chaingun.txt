ACTOR SDChaingun : Chaingun replaces Chaingun
{
	Weapon.BobStyle Alpha
	decal bulletchip
	States
	{
	Ready:
		PKCG A 1 A_WeaponReady
		Loop
	Deselect:
		PKCG A 1 A_Lower
		Loop
	Select:
		PKCG A 1 A_Raise
		Loop
	Fire:
		TNT1 A 0 A_PlaySound("weapons/chngun", CHAN_WEAPON)
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_FireBullets(5.6, 1, 1, 5, "BulletPuff")
		PKCG ABCD 1
		PKCG B 0 A_ReFire
		PKCG A 1 A_Weaponready
		PKCG BB 1 A_Weaponready
		PKCG CCC 1 A_Weaponready
		PKCG DDD 1 A_Weaponready
		Goto Ready
	Flash:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Recoil")==1, "KickFlash")
		TNT1 A 0 A_Jump(256,1,4)
		PKCF A 1 Bright A_Light2
		PKCF B 1 Bright A_Light1
		TNT1 A 2 A_Light0
		Goto LightDone
		PKCF C 1 Bright A_Light2
		PKCF D 1 Bright A_Light1
		TNT1 A 2 A_Light0
		Goto LightDone
	KickFlash:
		TNT1 A 0 A_Jump(256,1,7)
		PKCF A 1 Bright A_Light2
		TNT1 A 0 A_SetPitch(pitch+0.3)
		PKCF B 1 Bright A_Light1
		TNT1 A 0 A_SetPitch(pitch-0.1)
		TNT1 A 2 A_Light0
		TNT1 A 0 A_SetPitch(pitch-0.2)
		Goto LightDone
		PKCF C 1 Bright A_Light2
		TNT1 A 0 A_SetPitch(pitch+0.3)
		PKCF D 1 Bright A_Light1
		TNT1 A 0 A_SetPitch(pitch-0.1)
		TNT1 A 2 A_Light0
		TNT1 A 0 A_SetPitch(pitch-0.2)
		Goto LightDone
	}
}
