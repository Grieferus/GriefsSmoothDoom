ACTOR SDPlasmaRifle : PlasmaRifle replaces PlasmaRifle
{
	Weapon.BobStyle Alpha
	States
	{
	Fire:
		PLSF A 1 Bright A_FirePlasma
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Recoil")==0, "+2")
		TNT1 A 0 A_QuakeEx(1,1,1,2,0,10,"misc/null",QF_SCALEDOWN)
		PLSF BC 1 Bright
		TNT1 A 0 A_ReFire(+1)
		Goto Cooldown
		PLSF D 1 Bright A_FirePlasma
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Recoil")==0, "+2")
		TNT1 A 0 A_QuakeEx(1,1,1,2,0,10,"misc/null",QF_SCALEDOWN)
		PLSF CB 1 Bright
		TNT1 A 0 A_ReFire("fire")
		Goto Cooldown
	Cooldown:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gloves")==1, "BLCooldown")
		PKPL A 1 A_ZoomFactor(1.0, ZOOM_NOSCALETURNING)
		PKPL BCD 1
		PKPL E 9
		PKPL HDGCFBA 1
		Goto Ready
	BLCooldown:
		PKPL A 1 A_ZoomFactor(1.0, ZOOM_NOSCALETURNING)
		BKPL BCD 1
		BKPL E 9
		BKPL HDGCFB 1
		PKPL A 1
		Goto Ready
	Flash:
		TNT1 A 1 A_Light1
		Goto LightDone
		TNT1 A 1 A_Light1
		Goto LightDone
	}
}

ACTOR SDPlasmaBall : PlasmaBall replaces PlasmaBall
{
decal plasmascorchlower
States
	{
	Spawn:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_FancyPuff")==1, "FancySpawn")
		BLPL ABCD 2 Bright
		Loop
	FancySpawn:
		BLPL ABCD 2 Bright A_SpawnItemEx("BlueSparkle", random(-20, 20), random(-20, 20), random(0, 20))
		Loop
	Death:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_FancyPuff")==1, "FancyDeath")
		SPLS ABCDEFGHIJKL 2 Bright
		Stop
	FancyDeath:
		TNT1 AA 0 A_SpawnItemEx("BlueSparkle2", 0, 0, 0, random(1,3),random(-1,1),random(0,-2))
		TNT1 AA 0 A_SpawnItemEx("BlueSparkle2", 0, 0, 0, random(1,2),random(-3,3),random(-5,3))
		SPLS ABCDEFGHIJKL 2 Bright
		Stop
	}
}
