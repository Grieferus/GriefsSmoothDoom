ACTOR SDShotgun : Shotgun replaces Shotgun
{
	Weapon.BobStyle Alpha
	decal bulletchip
	States
	{
	Ready:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gloves")==1, "BLReady")
		PKSG A 1 A_WeaponReady
		Loop
	Deselect:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gloves")==1, "BLDeselect")
		PKSG A 1 A_Lower
		Loop
	Select:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gloves")==1, "BLSelect")
		PKSG A 1 A_Raise
		Loop
	Fire:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gloves")==1, "BLfire")
		PKSG A 3
		TNT1 A 0 A_FireBullets(5.6, 1, 7, 5, "BulletPuff")
		TNT1 A 0 A_PlaySound("weapons/shotgf",CHAN_WEAPON)
		TNT1 A 0 A_GunFlash
		PKSG AAOPQPO 1
		PKSG BCDEFGHIJ 1
		PKSG KL 2
		PKSG M 3
		PKSG LKJIHGFED 1
		PKSG CB 2
		PKSG A 5 A_ReFire
		Goto Ready
	Flash:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Recoil")==1, "KickFlash")
		SHTF B 1 Bright A_Light2
		SHTF A 1 Bright A_Light1
		Goto LightDone
	KickFlash:
		SHTF B 1 Bright A_Light2
		SHTF A 1 Bright A_Light1
		TNT1 A 1 a_setpitch(pitch-0.4)
		TNT1 A 1 a_setpitch(pitch-0.4)
		TNT1 A 1 a_setpitch(pitch-0.2)
		TNT1 A 1 a_setpitch(pitch-0.2)
		TNT1 A 1 a_setpitch(pitch+0.2)
		TNT1 A 1 a_setpitch(pitch+0.2)
		TNT1 A 1 a_setpitch(pitch+0.2)
		TNT1 A 1 a_setpitch(pitch+0.2)
		TNT1 A 1 a_setpitch(pitch+0.2)
		TNT1 A 1 a_setpitch(pitch+0.2)
		Goto LightDone
	BLReady:
		BKSG A 1 A_WeaponReady
		Loop
	BLDeselect:
		BKSG A 1 A_Lower
		Loop
	BLSelect:
		BKSG A 1 A_Raise
		Loop
	BLFire:
		BKSG A 3
		TNT1 A 0 A_FireBullets(5.6, 1, 7, 5, "BulletPuff")
		TNT1 A 0 A_PlaySound("weapons/shotgf",CHAN_WEAPON)
		TNT1 A 0 A_GunFlash
		BKSG AAOPQPO 1
		BKSG BCDEFGHIJ 1
		BKSG KL 2
		BKSG M 3
		BKSG LKJIHGFED 1
		BKSG CB 2
		BKSG A 5 A_ReFire
		Goto Ready
	}
}
