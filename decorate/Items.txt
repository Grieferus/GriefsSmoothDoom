ACTOR SDExplosiveBarrel : ExplosiveBarrel replaces ExplosiveBarrel
{
	PainChance 256
	States
	{
	Spawn:
		BAR1 ABCDEFGHIJKL 3
		Loop
	Pain:
		SEXP B 7
		BAR1 I 2
		BAR1 A 3
		SEXP B 5
		Goto Spawn
	Death:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Rumble")==0, "+3")
		TNT1 A 0 A_Quake (2,3,0,600,0)
		TNT1 A 0 A_Quake (1,7,0,1000,0)
		TNT1 A 0 A_JumpIf(GetCVAR("SD_FancyPuff")==1, "FancyDeath")
		BAR1 I 5 Bright
		SEXP B 5 A_Scream
		SEXP CD 3 Bright
		SEXP E 3 Bright A_Explode
		SEXP FGHIJK 2 Bright
		TNT1 A 1050 Bright A_BarrelDestroy
		TNT1 A 5 A_ReSpawn
	FancyDeath:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Rumble")==0, "+3")
		TNT1 A 0 A_Quake (2,3,0,600,0)
		TNT1 A 0 A_Quake (1,7,0,1000,0)
		BAR1 I 5 Bright
		SEXP B 5 A_Scream
		SEXP CD 3 Bright
		SEXP E 3 Bright A_Explode
		TNT1 AAAAAAAAAAAAAAAAAAAA 0 A_SpawnItemEx("ExplosionParticles",random(-1,1),random(-1,1),random(-1,1),random(-4,4),random(-4,4),random(0,6),random(0,360))
		TNT1 AA 0 A_SpawnItemEx("SDSmoke",random(-1,1),random(-1,1),random(-1,1),random(-4,4),random(-4,4),random(0,6),random(0,360))
		SEXP FGHIJK 2 Bright
		TNT1 A 1050 Bright A_BarrelDestroy
		TNT1 A 5 A_ReSpawn
	}
}
