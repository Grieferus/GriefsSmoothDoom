ACTOR SDZombieman : Zombieman replaces Zombieman
{
	states
	{
	Spawn:
		PSSS AB 10 A_Look
		Loop
	See:
		POSW ABCDEFGH 4 A_Chase
		Loop
	Missile:
		PSSA A 5 A_FaceTarget
		PSSA B 2 A_FaceTarget
		POSS E 3 A_FaceTarget
		POSS F 3 Bright A_PosAttack
		POSS E 3
		PSSA BA 4
		PSSA A 2
		Goto See
	Pain:
		STRO A 3
		STRO B 3 A_Pain
		Goto See
	Death.GibbyHaynes:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_SSGGore")==0, "Death")
		TNT1 A 0 A_JumpIfCloser(128, "XDeath")
		Goto Death
	Death.Saw:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_SawGore")==0, "Death")
		Goto XDeath
	Death:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_MoreDeaths")==0, "Death1")
		TNT1 A 0 A_Jump(256, "Death1", "Death2", "Death3")
	Death1:
		5OSS A 3 A_Scream
		5OSS B 3 A_NoBlocking
		5OSS CDEFG 2
		5OSS H 3
		5OSS I -1
		Stop
	Death2:
		PSD1 A 2
		PSD1 B 2 A_Scream
		PSD1 C 3 A_NoBlocking
		PSD1 DEFGHIJKLM 2
		PSD1 N -1
		Stop
	Death3:
		PSD2 A 3
		PSD2 B 2 A_Scream
		PSD2 C 2 A_NoBlocking
		PSD2 DEF 2
		PSD2 G 3
		PSD2 HI 2
		PSD2 J -1
		stop
	XDeath:
		TGIB AB 2
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gore")==0, +2)
		TNT1 A 0 A_SpawnItemEx("GoreController")
		TGIB C 2 A_XScream
		TGIB D 2
		TGIB E 2 A_NoBlocking
		TGIB FGHIJKLMNOP 2
		TGIB Q -1
		Stop
	Death.Massacre:
		TGIB AB 2
		TGIB C 2 A_XScream
		TGIB D 2
		TGIB E 2 A_NoBlocking
		TGIB FGHIJKLMNOP 2
		TGIB Q -1
		Stop
	Raise:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_MoreDeaths")==0, "Raise1")
		TNT1 A 0 A_Jump(256, "Raise1", "Raise2", "Raise3")
	Raise1:
		5OSS IHGFEDCBA 3
		Goto See
	Raise2:
		PSD1 NMLKJIHGFEDCBA 3
		Goto See
	Raise3:
		PSD2 JIHGFEDCBA 3
		Goto See
	}
}
