ACTOR SDShotgunGuy : ShotgunGuy replaces ShotgunGuy
{
	States
	{
	Spawn:
		SARR AB 10 A_Look
		loop
	See:
		SPOS ABCDEFGH 3 A_Chase
		Loop
	Missile:
		SPSR A 1
		SPSR AC 2 A_FaceTarget
		SPSR E 5 A_FaceTarget
		SPSR F 3 Bright A_SposAttackUseAtkSound
		SPSR E 2
		SPSR CAABBA 3
		Goto See
	Pain:
		SSPA A 3
		SSPA B 3 A_Pain
		Goto See
	Death.GibbyHaynes:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_SSGGore")==0, "Death")
		TNT1 A 0 A_JumpIfCloser(128, "XDeath")
		Goto Death
	Death.Saw:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_SawGore")==0, "Death")
		Goto XDeath
	Death:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_MoreDeaths")==0, "Death1")
		TNT1 A 0 A_Jump(256, "Death1", "Death2", "Death3")
	Death1:
		SOSS Q 3
		SOSS R 3 A_Scream
		SOSS S 3 A_NoBlocking
		SOSS TUVWX 2
		SOSS Y -1
		Stop
	Death2:
		SO2S A 3
		SO2S B 3 A_Scream
		SO2S C 3 A_NoBlocking
		SO2S DEFGHIJKL 3
		SO2S M -1
		Stop
	Death3:
		SPOO A 3
		SPOO B 3 A_Scream
		SPOO C 3 A_NoBlocking
		SPOO DEFGHIJKLM 3
		SPOO N -1
		Stop
	XDeath:
		SGIB AB 2
		TNT1 A 0 A_JumpIf(GetCVAR("SD_Gore")==0, +2)
		TNT1 A 0 A_SpawnItemEx("GoreController")
		SGIB C 2 A_XScream
		SGIB D 2
		SGIB E 2 A_NoBlocking
		SGIB FGHIJKLMNOP 2
		SGIB Q -1
		Stop
	Death.Massacre:
		SGIB AB 2
		SGIB C 2 A_XScream
		SGIB D 2
		SGIB E 2 A_NoBlocking
		SGIB FGHIJKLMNOP 2
		SGIB Q -1
		Stop
	Raise:
		TNT1 A 0 A_JumpIf(GetCVAR("SD_MoreDeaths")==0, "Raise1")
		TNT1 A 0 A_Jump(256, "Raise1", "Raise2", "Raise3")
	Raise1:
		SOSS YXWVUTSRQ 3
		Goto See
	Raise2:
		SO2S MLKJIHGFEDCBA 3
		Goto See
	Raise3:
		SPOO NMLKJIHGFEDCBA 3
		Goto See
		}
}
